for (i in seq_along(ids)) {
  id <- ids[i]
  f <- paste0("data/", id, ".gexf")

  x <- read.gexf(f)
  g <- gexf.to.igraph(x)
  g <- igraph::simplify(g, remove.multiple = FALSE, remove.loops = TRUE)

  deg <- degree(g)

  g_cl <- tryCatch({
    tmp <- sample_degseq(deg, method = "vl")
    igraph::simplify(tmp, remove.multiple = TRUE, remove.loops = TRUE)
  }, error = function(e) {
    message("Failed on movie id: ", id)
    return(NULL)
  })

  if (is.null(g_cl)) next

  res[[i]] <- data.frame(
    id = id,
    movie = movie_names[i],

    real_nodes = vcount(g),
    real_edges = ecount(g),
    real_density = edge_density(g),
    real_transitivity = transitivity(g, type = "global"),
    real_avg_transitivity = transitivity(g, type = "average"),
    real_triangles = sum(count_triangles(g)) / 3,

    cl_nodes = vcount(g_cl),
    cl_edges = ecount(g_cl),
    cl_density = edge_density(g_cl),
    cl_transitivity = transitivity(g_cl, type = "global"),
    cl_avg_transitivity = transitivity(g_cl, type = "average"),
    cl_triangles = sum(count_triangles(g_cl)) / 3
  )
}
